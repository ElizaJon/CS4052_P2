package modelChecker;

import formula.PathTree;
import formula.stateFormula.StateFormula;
import model.Model;
import model.State;

public class SimpleModelChecker implements ModelChecker {
    PathTree root;
    @Override
    public boolean check(Model model, StateFormula constraint, StateFormula query) {
        // TODO Auto-generated method stub
        System.out.println(constraint.toString());
        System.out.println(query.toString());

        System.out.println("\nNewer stuff");

        State[] allStates = model.getStates();
        root = new PathTree("");
        PathTree leftNode = new PathTree("");
        root.leftTree = leftNode;
        State[] resultStates = constraint.getStates(allStates, model, leftNode);
        System.out.println("Resulting states:");
        printStates(resultStates);
        printTree(root);
        boolean result = setOfStatesHasInitialState(resultStates);
        root.setModelHolds(result);
        System.out.println("Has initial states: " + result);
        getTrace();
        return result;
    }

    private boolean setOfStatesHasInitialState(State[] resultStates){
        for(int i = 0; i < resultStates.length; i++){
            if(resultStates[i].isInit()){
                return true;
            }
        }
        return false;
    }

    public static void printStates(State[] result){
        for(int i = 0; i < result.length; i++) {
            System.out.print(result[i].getName() + " ");
        }
        System.out.println();
    }

    @Override
    public String[] getTrace() {
        printFalseSide(root);
        return null;
    }

    private void printFalseSide(PathTree root){
        if(root != null && root.getFormulaPart().equals(" A ")){
            printTreeForA(root);
            return;
        }
        if (root == null || root.getModelHolds()){
            return;
        }
        System.out.print(root.getFormulaPart());
        System.out.print(root.getModelHolds() + " ");
        printStates(root.getAcceptedStates());
        printFalseSide(root.getLeftTree());
        printFalseSide(root.getRightTree());
    }

    private void printTreeForA(PathTree root){
        if(root == null){
            return;
        }
        System.out.print(root.getFormulaPart());
        System.out.print(root.getModelHolds() + " ");
        printStates(root.getAcceptedStates());
        printTreeForA(root.getLeftTree());
        printTreeForA(root.getRightTree());
    }

    private void printTree(PathTree root){
        if(root == null){
            return;
        }
        printTree(root.getLeftTree());
        System.out.print(root.getFormulaPart());
        printStates(root.getAcceptedStates());
        printTree(root.getRightTree());
    }

}
